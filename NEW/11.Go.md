# 6. Variables
-------------
	var [variable name] [data type] = [value]
	
	Integer:
	var integer1 int = 1 //stores any integer
	var unsignedInteger2 uint = 8 //only stores positive values
	var signedInteger3 int8 = 250 //only store 8bit values (can use 8,16,32,64)

	Float (decimal):
	var float1 float32 = 28.790 //stores 32 bit float value (can use 32,64)

	Boolean (true/false):
	var trueOrFalse bool = true //stores either true or false (can use true,false as values)

	String:
	var myName string = "john"

	Const (value cannot be changed) :
	format : const [variable name] [data type] = [value]
	eg: const constantNumber int = 300

	Short declaration:
	format : [variablename] := [value]
	eg: myNumber := 298

# 8. Arrays
--------------------------
```
-- Simple collection datatype
-- Fixed size
-- Limited to only one Datatype
```
declaring :

	format : var arrayname [array_size] data_type
	eg: var fruits [5] strings

usage :

	var fruits [5]string
	fruits = [5]string{"apple", "orange", "grape", "strawberry", "banana"} //assign values into array
	fmt.Println("array : ", fruits)                                        //print whole array
	fmt.Println("2nd fruit", fruits[1])                                    //access element from array with index no
	fruits[1] = "avocado"                                                  //modify element of array using index no
	fmt.Println("modified array : ", fruits)                               //print the modified array
	var length_of_array = len(fruits)                                      //get the length of the array
	fmt.Println("the length of array : ", length_of_array)                 // prints the length of fruits array
	
> result :
------------------------
	array :  [apple orange grape strawberry banana]
	2nd fruit orange
	modified array :  [apple avocado grape strawberry banana]
	the length of array :  5

# 9. Slices
--------------------------
```
-- Reference collection datatype
-- variable size, so we can add/remove items
-- Limited to only one Datatype
```

declaring :

	format : var arrayname [] data_type
	eg: var fruits [] strings

usage :

	package main
	
	import (
		"fmt"
		"slices"
	)
	
	func main() {
		var fruits []string
		fruits = []string{"apple", "orange", "grape"}   //assign values into slice
		fmt.Println(fruits)                             //print whole array
		fruits = append(fruits, "guava")                //append new value into slice
		fmt.Println(fruits)                             //print modified slice
		fruits = append(fruits, "strawberry", "banana") //append multiple values at once
		fmt.Println(fruits)                             //print modified slice
		fruits = slices.Delete(fruits, 1, 3)            //remove elements in slice from ind no 1 to 2
		fmt.Println(fruits)                             //print modified slice
	}

> result :
------------------------
	[apple orange grape]
	[apple orange grape guava]
	[apple orange grape guava strawberry banana]
	[apple guava strawberry banana]

# 10. Heterogeneous Slice

declaring :

# 11. Maps
--------------------------
```
-- Reference datatype
-- Key-Value pairs
-- Dynamic size
-- Limited to only one Datatype
```

declaring :

	format : var map_name map[key_type] value_type
	eg: var user map[string] string

assigning :

	format : map_name = map[key_datatype] value_datatype{ "key1":"value1","key2":"value2"}
	eg: users = map[string]string{"name": "John", "country": "Netherland", "gender": "male"}

usage :

	var user map[string]string
	user = map[string]string{"name": "John", "country": "Netherland", "gender": "male"}
	fmt.Println(user)                                     //print the whole map
	fmt.Println("the name is : ", user["name"])           //print only the value of given key
	user["country"] = "korea"                             //modify the value of key by key name
	fmt.Println("the modified field : ", user["country"]) //print modified value
	user["age"] = "21"                                    //add new key
	fmt.Println("age : ", user["age"])                    //print the new key

> result :

	map[country:Netherland gender:male name:John]
	the name is :  John
	the modified field :  korea
	age :  21

**Map of Slices**

	var shop map[string][]string
	shop = map[string][]string{"fruits": {"apple", "banana"}, "vegetables": {"tomato", "pumkin"}}
	fmt.Println("fruit : ", shop["fruits"][0])                 //print the element of slice oof key
	shop["meat"] = []string{"fish", "chicken", "beef", "pork"} //add new key value pair of slice
	fmt.Println("new value : ", shop["meat"])                  //print the new value of key
	delete(shop, "vegetables")                                 //remove a key and its value
	fmt.Println("new shop content : ", shop)                   //print modified map

> result :

	fruit :  apple
	new value :  [fish chicken beef pork]
	new shop content :  map[fruits:[apple banana] meat:[fish chicken beef pork]]

**Slice of Maps**

	import (
		"fmt"
		"slices"
	)
	
	func main() {
		var students = []map[string]string{
			{"name": "john", "age": "21"},
			{"name": "george", "age": "27"},
			{"name": "alwin", "age": "18"},
		}
	
		fmt.Println(students)                              //print the whole slice
		fmt.Println("1st student : ", students[0]["name"]) //access map value from array using key
		students[1]["name"] = "morgan"                     //modify the value of key of map in slice
		fmt.Println(students)                              //print the modified slice
		students = slices.Delete(students, 2, 3)           //remove a map in slice from index no 2 till 3
		fmt.Println(students)                              //print the modified slice
	}

> result :

	[map[age:21 name:john] map[age:27 name:george] map[age:18 name:alwin]]
	1st student :  john
	[map[age:21 name:john] map[age:27 name:morgan] map[age:18 name:alwin]]
	[map[age:21 name:john] map[age:27 name:morgan]]

# 12. Struct
--------------------------
```
-- it is the only heterogeneus (can contain multiple data types) data type in go 
-- custom type
-- Basic building block for the OOPs in golang
```

declaring :

	format : type struct_name struct { 
			variable1 datatype
			variable2 datatype
		} 

	eg : type Student struct {
			name string
			age int
			weight float32
			passed bool
	}

assigning :

	var student1 = Student{name: "student_name",age:age_int,weight:student_weight,passed:true_or_false}

usage :

	package main
	
	import "fmt"
	
	type Student struct {
		name   string
		age    int
		weight float32
		passed bool
	} //declare struct model
	
	func main() {
		var student1 = Student{name: "John", age: 15, weight: 45.23, passed: true}   //assign values
		var student2 = Student{name: "Peter", age: 20, weight: 52.88, passed: false} //assign values
		fmt.Println(student1)                                                        //print whole student1 content
		fmt.Println(student2.name)                                                   //print name field of student2 struct variable
		
	}

> result :

	{John 15 45.23 true}
	{Peter 20 52.88 false}

**Slices of Structs**

	package main
	
	import "fmt"
	
	type Student struct {
		name   string
		age    int
		weight float32
		passed bool
	}
	
	func main() {
		var student1 = Student{name: "John", age: 15, weight: 45.23, passed: true}   //assign values
		var student2 = Student{name: "Peter", age: 20, weight: 52.88, passed: false} //assign values
	
		var students = []Student{student1, student2}
		fmt.Println(students)
	}

> result :

	[{John 15 45.23 true} {Peter 20 52.88 false}]

**Struct with Slices**

	package main
	
	import "fmt"
	
	type Student struct {
		name      string
		languages []string
	}
	
	func main() {
		var student1 = Student{name: "John", languages: []string{"english", "spanish", "french"}}
		fmt.Println("name : ", student1.name)
		fmt.Println("languages : ", student1.languages)
	}

> result :

	name :  John
	languages :  [english spanish french]

**Function with Arguments and Parameters**

	package main
	
	import "fmt"
	
	func find_sum(num1 int, num2 int) {
		var sum = num1 + num2
		fmt.Printf("The sum of %d and %d is %d", num1, num2, sum)
	}
	
	func main() {
		find_sum(7, 8)
	}

> result :

	The sum of 1 and 2 is 3 

**Function with Return**

	package main
	
	import "fmt"
	
	func find_sum(num1 int, num2 int) int {
		return num1 + num2
	}
	
	func main() {
		num1 := 2
		num2 := 7
		fmt.Printf("the sum of %d and %d is %d", num1, num2, find_sum(num1, num2))
	}

> result :

	the sum of 2 and 7 is 9
