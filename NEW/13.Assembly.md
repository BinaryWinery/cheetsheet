# Print Hello world

    section .data
        variable1 db "Hello world",0xA
        variable1_length equ $- variable1
    
    section .text
        global _start
    
    _start:
        ;PRINTING HELLO WORLD
        mov rax,1
        mov rdi,1
        mov rsi,variable1
        mov rdx,variable1_length
        syscall    

        ;EXITING PROGRAM
        mov rax,60
        mov rdi,0
        syscall
