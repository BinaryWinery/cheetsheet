# Print Hello world

    section .data
        variable1 db "Hello world",0xA
        variable1_length equ $- variable1
    
    section .text
        global _start
    
    _start:
        ;PRINTING HELLO WORLD
        mov rax,1
        mov rdi,1
        mov rsi,variable1
        mov rdx,variable1_length
        syscall    

        ;EXITING PROGRAM
        mov rax,60
        mov rdi,0
        syscall

Result  
> 
     Hello world

# Get User Input

    section .data
        prompt db "Enter your name : " ; PROMPT LABEL FOR ENTERING INPUT
        prompt_length equ $- prompt
    
        greetMessage db "Hello "
        greetMessage_length equ $- greetMessage
    
    section .bss
        name resb 64 ;RESERVER 64 BYTES OF MEMMORY FOR INPUT
    
    section .text
        global _start
    
    _start:
    
        ; PRINT PROMPT 
        mov rax,1
        mov rdi,1
        mov rsi,prompt
        mov rdx,prompt_length
        syscall 
    
        ; GET INPUT
        mov rax,0
        mov rdi,0
        mov rsi,name
        mov rdx,64
        syscall
    
        ; PRINT GREET MESSAGE
        mov rax,1
        mov rdi,1
        mov rsi,greetMessage
        mov rdx, greetMessage_length
        syscall
    
    
        ; PRINT THE VALUE
        mov rax,1
        mov rdi,1
        mov rsi,name
        mov rdx,64
        syscall
    
        ; EXIT THE PROGRAM
        mov rax,60
        mov rdi,0
        syscall
